TERMINAL

cls  = limpa terminal
cd = navega entre as pastas
dir = lista todas as pastas
tab = autocompleta 
mkdir + nomedapasta = cria pasta
echo hello > hello.txt  (escreve hello dentro do arquivo hello.txt , caso hello.txt não exista ele cria também)
del "nome da pasta" = deleta todos os arquivos dentro da pasta (*obs , nao exclui a pasta);
rmdir "nome da pasta" /S /Q  = remove o diretório e todos os arquivos dentro dele .
